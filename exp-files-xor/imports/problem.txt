$qFunctions = f.allNamed(fs = m.sKeyMap(
  keys = [behavior; avgLastBoolScore; avgLastUnlimitedScore; increase; consistency];
  values = [
    f.identity();
    ds.e.sxor.avgScore(
      rewardType = boolean;
      caseIndexes = [100:1:139]   % the last 40 cases
    );
    ds.e.sxor.avgScore(
      rewardType = unlimited;
      caseIndexes = [100:1:139]   % the last 40 cases
    );
    ds.e.sxor.avgScoreDelta(
      rewardType = unlimited;
      firstIndexes = [-4:1:-1];  % the last 4 cases
      secondIndexes = [0:1:3]     % the first 4 cases
    );
    ds.e.sxor.avgScoreVariation(
      rewardType = unlimited;
      indexes = [-8:1:-1]         % the last 8 cases
    )
  ]
))

$problemHomoSame = ea.p.simToTo(

    cases = $homoSameCases;
